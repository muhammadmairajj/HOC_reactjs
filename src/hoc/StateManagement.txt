// A Higher-Order Component (HOC) that manages and shares state:

import React, { useState } from 'react';

// Higher-Order Component for state management
const withState = (WrappedComponent) => {
    return (props) => {
        const [count, setCount] = useState(0);

        const increment = () => setCount(count + 1);

        return <WrappedComponent count={count} increment={increment} {...props} />;
    };
};

// You can make separate component of Counter
const Counter = ({ count, increment }) => (
    <div>
        <p>Count: {count}</p>
        <button onClick={increment}>Increment</button>
    </div>
);

const StatefulCounter = withState(Counter);

const App = () => {
    return <StatefulCounter />;
};

export default App;
